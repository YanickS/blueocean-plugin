export const RESULTS = keymirror({
    UNKNOWN: null,
    SUCCESS: null,
    FAILURE: null,
});
 
export const STATES = keymirror({
    RUNNING: null,
    PAUSED: null,
    FINISHED: null,
});
 
function isRunningNode(item) {
    return item.state === STATES.RUNNING || item.state === STATES.PAUSED;
}
